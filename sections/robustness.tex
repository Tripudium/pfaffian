\section{Robustness}\label{sec:robustness}
Consider a data space $\mX$ with a norm $\|\cdot\|$ (for example, $\R^d$ with the Euclidean norm) and a map $F\colon \mX \to \R^k$, 
where $F=(f_1,\dots,f_k)$. We interpret this map as a classifier, with $f_i(\vct{x})$ representing the likelihood that $\vct{x}$ belongs to class $i$. Thus a data point $\vct{x}\in \mX$ is assigned to class $j$ if $f_j(\vct{x})>f_i(\vct{x})$ for all $i\neq j$, with an arbitrary tie break if $f_i(\vct{x})=f_j(\vct{x})$. We denote the class to which $\vct{x}$ is assigned to by $\hat{j}(\vct{x})$.
For $i\neq j$, define the functions
$g_{ij} := f_j-f_i$. The map $F$ induces a subdivision of $\mX$ into $k$ regions
\begin{equation*}
 C_j = \{\vct{x}\in \mX \colon g_{ij}(\vct{x}) \geq 0 \text{ for } i\neq j\}, \quad \quad j\in \{1,\dots,k\}.
\end{equation*}
For convenience, and because this is the practically relevant setting, we assume that the functions $g_{ij}$ are continuous and that the sets $C_j$ have a non-empty interior. The sets $C_j$ are thus closed and the interior $\inter(C_j)$ is the set of $\vct{x}$ that are unequivocally assigned to class $j$. 
The boundary of $C_j$ and the decision boundary of the classification problem are given by
\begin{equation*}
 \Sigma_j = \bigcup_{i\neq j} (C_j\cap C_i), \quad j\in \{1,\dots,k\} \quad \text{ and }  \quad \Sigma = \bigcup_j \Sigma_j,
\end{equation*}
respectively. 

For each $j\in \{1,\dots,k\}$ and $\vct{x}\in \mX$, define the distance 
\begin{equation*}
 \Delta_j(\vct{x}) = \inf \{ \|\vct{x}-\vct{y}\| \colon \vct{y}\not\in C_j\} =: \dist(\vct{x},\mX \backslash C_j).
\end{equation*}
It follows that $\Delta_j(\vct{x})=0$ if and only if $\vct{x}\in \mX\backslash C_j$, and for $\vct{x}\in C_j$, $\Delta_j(\vct{x})$ is the distance to misclassification. For arbitrary $\vct{x}$, the distance to misclassification (or robustness certificate) is given by
\begin{equation}
 \Delta(\vct{x}) := \sum_{i=1}^k \Delta_i(\vct{x}). 
\end{equation}
This distance can be defined alternatively as $\Delta(\vct{x}) = \inf \{ \|\vct{x}-\vct{y}\| \colon \vct{y}\in \Sigma\} = \dist(\vct{x},\Sigma)$.
Assuming that $\vct{x}\in \mX\backslash \Sigma$ has a unique closest point on $\Sigma$, define
\begin{equation*}
\vct{r}(\vct{x}) := \argmin_{\vct{r}}\{\|\vct{r}\| \mid \vct{x}+\vct{r}\in \Sigma\} = \argmin_{\vct{r}} \{\|\vct{r}\| \mid \hat{j}(\vct{x}+\vct{r})\neq \hat{j}(\vct{x})\}.
\end{equation*} 
In particular, in this case we have $\Delta(\vct{x})=\|\vct{r}(\vct{x})\|$. 

It is reasonable to measure the distance to misclassification relative to the size of $\vct{x}$. In analogy to the theory of conditioning in numerical analysis 
and optimization~\cite{Condition}, we can define the condition of $\vct{x}$.

\begin{definition}
The condition of the classification map $F$ at $\vct{x}\in \mX\backslash \Sigma$ is defined as
\begin{equation*}
  \mC(\vct{x}) := \frac{\|\vct{x}\|}{\Delta(\vct{x})}. 
\end{equation*}
\end{definition}

The robustness, and the associated notion of condition number, quantify the sensitivity of a classification problem to worst-case perturbations of an input $\vct{x}$, and is therefore referred to as adversarial robustness. A different problem arises if we ask for the sensitivity of a classifier to random perturbations, rather than targeted, adversarial perturbations. For absolutely continuous probability distributions, the probability that a random perturbation of size $\Delta(\vct{x})$ will lead to misclassification is zero. For a random perturbation of size $\eta>\Delta(\vct{x})$, however, this probability can be non-negligible. This motivates Definition~\ref{def:randrob}.
In what follows, we denote by $\mathbb{S}:=S^{d-1}$ the unit sphere in $\R^d$ and we write $\Prob_{\vct{v}\in \mathbb{S}}(\vct{v}\in A)=\mu_{\mathrm{unif}}(A)$, where
$\mu_{\mathrm{unif}}$ is the uniform probability measure on $\mathbb{S}$. We can get a uniformly distributed $\vct{v}$ on $\mathbb{S}$ by taking a Gaussian vector $\vct{g}$ and setting $\vct{v}=\vct{g}/\|\vct{g}\|_2$. 

\begin{definition}\label{def:randrob}
For $\epsilon\in [0,1)$, define the (uniform) $\epsilon$-robustness of $f\colon \mX\to \R^k$ at $\vct{x}\in \mX$ as
\begin{equation*}
  \Delta_{\epsilon}(\vct{x}) = \Delta_{\mathrm{unif},\epsilon}(\vct{x}) := \inf \{\eta \mid \Prob_{\vct{v}\in \mathbb{S}}(\hat{j}(\vct{x}+\eta \vct{v})\neq \hat{j}(\vct{x}))> \epsilon\}.
\end{equation*}
\end{definition}

Clearly, $\Delta_0(\vct{x})=\Delta(\vct{x})$. 
Note that if $\vct{x}\not\in \Sigma$ and $C_{\hat{j}(\vct{x})}$ is bounded, then $\Delta_{\epsilon}$ is finite, but this need not be the case in general. 
Under certain circumstances, one can interpret $\Delta_{\epsilon}(\vct{x})$ as a quantile function. More precisely, assume $C_{\hat{j}(\vct{x})}$ is star-shaped, and consider the function
$\Delta(\vct{x};\vct{v})=\inf\{\eta \colon \hat{j}(\vct{x}+\eta \vct{v})\neq \hat{j}(\vct{x})\}$. Then $\hat{j}(\vct{x}+\eta \vct{v})\neq \hat{j}(\vct{x})$ is equivalent to
$\Delta(\vct{x};\vct{v})\leq \eta$, the function $\eta\mapsto \Prob(\Delta(\vct{x};\vct{v})\leq \eta)$ is monotonically increasing, and 
\begin{equation*}
  \Delta_{\epsilon}(\vct{x}) = \inf\{ \eta \colon \Prob_{\vct{v}\in \mathbb{S}}(\Delta(\vct{x};\vct{v})\leq \eta)> \epsilon\}, 
\end{equation*} 
which is the definition of a quantile function. This definition is reminiscent of the definition of weak condition number in~\cite{}.

\begin{definition}
Let $\delta\in [0,1)$. Then the $\delta$-weak condition number of the classification problem $F\colon \mX\to \R^k$ at $\vct{x}\in \mX\backslash \Sigma$ is defined to be 
\begin{equation*}
  \mathcal{C}_\delta(\vct{x}) = \inf \left\{ \eta \colon \Prob\left(\frac{\|\vct{x}\|}{\Delta(\vct{x};\vct{v})}\right) \right\}
\end{equation*}
\end{definition}

\subsection{Probabilistic analysis}
When considering a distribution on data space, we can consider different settings:

\begin{enumerate}
\item Robustness of random data;
\item Robustness of a random perturbation of a given data point.
\end{enumerate}

In the classical analysis of condition numbers, these two cases correspond to average-case and smoothed analysis, as discussed in~\cite{Condition}. 
We look at both these cases in a unified way. The first 
natural question to ask is regarding the distribution of $\Delta(X)$, namely
\begin{equation}\label{eqn:main-problem}
  \Prob(\Delta(X) \leq \epsilon)
\end{equation}
for $\epsilon\geq 0$. 
As an example, $X$ could be a Guassian density centred around $\vct{x}$. Alternatively, we may restrict to a single class, say $j$, and consider the probability
\begin{equation*}
  \Prob(\Delta_j(X)\leq \epsilon).
\end{equation*}
The case $\epsilon=0$ then corresponds to the probability of misclassification. 

 We are interested in whether 
a typical problem is robust to adversarial perturbations, or more generally, whether a typical perturbation of a fixed data point is 
adversarially robust.\\

{\bf Problem 1} (Smoothed analysis) Given a data point $\vct{x}$ and a random perturbation $\vct{\delta}$, find a bound on
\begin{equation*}
  \Prob\{\mathcal{C}(X)>t\}
\end{equation*}
for some $t>0$ in terms of parameters of the neural network.\\
This problem is an instance of smoothed analysis of condition numbers, studied in detail in~\cite{Condition}. 

\subsection{Gaussian perturbations} In the literature on smoothed analysis it is common to consider Guassian perturbations. 

\begin{lemma}\label{lem:lem1}
Let $X$ be normal distributed around a point $\overline{\vct{x}}\in \R^{d}$ with variance $\sigma^2 \mathrm{Id}$. Assume that $\Sigma\subset \R^{d}$ is contained in an algebraic hypersurface of degree at most $D$. Then for $t\geq t_0=8\econst dD$,
\begin{equation*}
\Prob(\|X-\overline{\vct{x}}\|\cdot \dist(X,\Sigma)^{-1}\geq t) \leq \frac{8\econst dD}{t}.
\end{equation*}
\end{lemma}

\begin{proof}
The proof relies on a corresponding bound from~\cite{lotz2015volume} for uniform perturbations and extends this to Gaussian ones. 
Set $A:=\{\vct{x}\in \R^{d} \mid \|\vct{x}-\overline{\vct{x}}\|\cdot \dist(x,\Sigma)^{-1}\geq t\}$. 
Let $U_{s}$ be uniformly distributed on a closed ball $B(\overline{\vct{x}},s)$ around $\overline{\vct{x}}$ of radius $s$. Note that $U_{s}$ has density
\begin{equation}\label{eq:uniform-density}
  \frac{1}{\mathrm{vol}\ B(0,1) s^{d}}\cdot \onemtx\{\|x-\overline{\vct{x}}\|\leq s\}.
\end{equation}
For a Gaussian vector $X$ centred at $\overline{\vct{x}}$ and with covariance $\sigma^2\mathrm{Id}$, we then get
\begin{align*}
 \Prob(X\in A) &= \frac{1}{(2\pi\sigma^2)^{\frac{d}{2}}}\int_A \econst^{-\frac{\|\vct{x}-\overline{\vct{x}}\|^2}{2\sigma^2}} \ \diff{\vct{x}} = \frac{1}{(2\pi\sigma^2)^{\frac{d}{2}}}\int_A \left(\int_{\frac{\|\vct{x}-\overline{\vct{x}}\|}{\sigma}}^\infty s \econst^{-\frac{s^2}{2}} \ \diff{s}\right) \ \diff{\vct{x}}\\
 &= \frac{1}{(2\pi\sigma^2)^{\frac{d}{2}}}\int_{0}^{\infty} \left( \int_A  \onemtx\{ \|\vct{x}-\overline{\vct{x}}\|\leq \sigma s\} \  \diff{\vct{x}} \right) s \econst^{-\frac{s^2}{2}} \ \diff{s} \\
 &\stackrel{(1)}{=} \frac{\mathrm{vol}\ B(0,1)}{(2\pi)^{\frac{d}{2}}}\int_{0}^{\infty} \Prob{\|U_{\sigma s}-\overline{\vct{x}}\| \cdot \dist(U_{\sigma s},\Sigma)^{-1}\geq t} s^{d}\cdot s \econst^{-\frac{s^2}{2}} \ \diff{s}\\
 &\leq \frac{\mathrm{vol}\ B(0,1)}{(2\pi)^{\frac{d}{2}}}\int_{0}^{\infty} \Prob{\dist(U_{\sigma s},\Sigma)^{-1}\geq t/(\sigma s)} s^{d}\cdot s \econst^{-\frac{s^2}{2}} \ \diff{s}.
\end{align*}
where in (1) we used ~\eqref{eq:uniform-density} and the definition of the set $A$.
Starting from~\cite[Theorem 1.1]{lotz2015volume} in the special case of codimension one, we have
 \begin{align*}
 \Prob(\dist(U_{\sigma s},\Sigma)^{-1}\geq t/(\sigma s))
& \leq 4 \sum_{i=0}^N\binom{d}{i+1}\left(\frac{2D}{t}\right)^{i+1}\left(1+\frac{1}{t}\right)^{d-1-i}\\
&= 4(N+1) \int_{0}^{\frac{2D}{t}} \left(s+1+\frac{1}{ t}\right)^{d-1} \ \diff{s}
 \leq \frac{8d D}{ t} \left[1+\frac{1+2D}{t}\right]^{d-1}.
\end{align*}
In particular, if $t\geq t_0:=8\econst dD\geq (1+2D)(d-1)$, then
\begin{equation*}
  \left[1+\frac{1+2D}{t}\right]^{d-1} \leq \left[1+\frac{1}{d-1}\right]^{d-1} \leq \econst.
\end{equation*}
Hence,
\begin{equation*}
 \Prob(\|X-\overline{\vct{x}}\|\cdot \dist(\vct{x},\Sigma)^{-1} \geq t) \leq \frac{8\econst dD}{t} \cdot \frac{\mathrm{vol}\ B(0,1)}{(2\pi)^{\frac{d}{2}}}\int_{0}^{\infty} s^{d+1} \econst^{-\frac{s^2}{2}} \ \diff{s}\\
  = \frac{8\econst dD}{t}
\end{equation*}
for $t\geq t_0$. The last equality follows from the identity
\begin{equation*}
 \mathrm{vol} \ B(0,1) = \frac{\pi^{\frac{d}{2}}}{\Gamma\left(\frac{d}{2}+1\right)},
\end{equation*}
and writing out the Gamma function,
\begin{equation*}
 \Gamma\left(\frac{d}{2}+1\right) = \int_0^\infty t^{\frac{d}{2}} \econst^{-t} \ \diff{t} \stackrel{t=s^2/2}{=} \frac{1}{2^{\frac{d}{2}}} \int_0^\infty s^{d+1} \econst^{-\frac{s^2}{2}} \ \diff{s}.
\end{equation*}
This completes the proof.
\end{proof}
